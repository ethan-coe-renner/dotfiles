* Straight
[[https://github.com/radian-software/straight.el][Straight.el]] is used to install and manage packages. The following code bootstraps straight so it can be used in the rest of the configuration.

#+begin_src emacs-lisp
(defvar bootstrap-version)
(let ((bootstrap-file
       (expand-file-name
        "straight/repos/straight.el/bootstrap.el"
        (or (bound-and-true-p straight-base-dir)
            user-emacs-directory)))
      (bootstrap-version 7))
  (unless (file-exists-p bootstrap-file)
    (with-current-buffer
        (url-retrieve-synchronously
         "https://raw.githubusercontent.com/radian-software/straight.el/develop/install.el"
         'silent 'inhibit-cookies)
      (goto-char (point-max))
      (eval-print-last-sexp)))
  (load bootstrap-file nil 'nomessage))
(straight-use-package 'use-package)
#+end_src

* Emacs
This section contains basic configuration which does not depend on any packages.

#+begin_src emacs-lisp
(use-package emacs
  :custom
  ;; Scratch Buffer
  (initial-major-mode 'org-mode)
  (initial-scratch-message "")
  (inhibit-startup-screen t)

  ;; Require confirmation when killing emacs
  (confirm-kill-emacs 'y-or-n-p)

  ;; Set custom file location, this is loaded in :config
  (custom-file (locate-user-emacs-file "custom.el"))

  ;; Use visible bell instead of audible bell
  (visible-bell t)

  ;; Don't resize frame, let the WM do this
  (frame-inhibit-implied-resize t)

  ;; Scrolling
  (scroll-error-top-bottom t)
  (scroll-preserve-screen-position t)

  ;; Use view-mode for read-only buffers
  (view-read-only t)

  ;; Put all backups in .saves instead of the original directory
  (backup-directory-alist `(("." . ,(concat user-emacs-directory "backups"))))

  ;; Backup by copying is safer for e.g. links
  (backup-by-copying t)

  ;; Keep some numbered versions with version control
  (delete-old-versions t)
  (kept-new-versions 6)
  (kept-old-versions 2)
  (version-control t)

  ;; Determine how time is displayed
  (display-time-default-load-average nil)
  (display-time-day-and-date t)
  (display-time-24hr-format t)

  ;; Sentences can end with a single space
  (sentence-end-double-space nil)

  :init
  ;; Use 4 spaces instead of tabs
  (setq-default indent-tabs-mode nil)
  (setq-default tab-width 4)

  :config
  ;; Load ancillary configuration files
  (org-babel-load-file (locate-user-emacs-file "my.org"))
  (load custom-file :no-error-if-file-is-missing)

  ;; Remove unnecessary ui elements
  (menu-bar-mode -1)
  (scroll-bar-mode -1)
  (tool-bar-mode -1)

  ;; Show misc info in modeline
  (which-function-mode 1)
  (display-battery-mode 1)
  (display-time-mode)

  ;; Disable keybinding for goal columns
  (global-unset-key (kbd "C-x C-n")))

* Load Machine Specific Configuration
I put machine specific configuration (work vs personal) into ~~/git/elisp~. Load that if it exists. Configuration for this should exist in a ~configuration.org~ in that directory. Use ~org-babel-load-file~ to load that as well.

#+begin_src emacs-lisp
(setq machine-specific-configuration-directory (expand-file-name "~/git/elisp"))
(setq machine-specific-configuration-org (expand-file-name "configure.org" machine-specific-configuration-directory))

(when (file-directory-p machine-specific-configuration-directory)
    (add-to-list 'load-path machine-specific-configuration-directory)
    (when (file-exists-p machine-specific-configuration-org)
      (org-babel-load-file (expand-file-name machine-specific-configuration-org))))
#+end_src
