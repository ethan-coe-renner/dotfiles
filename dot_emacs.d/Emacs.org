* Straight
[[https://github.com/radian-software/straight.el][Straight.el]] is used to install and manage packages. The following code bootstraps straight so it can be used in the rest of the configuration.

#+begin_src emacs-lisp
(defvar bootstrap-version)
(let ((bootstrap-file
       (expand-file-name
        "straight/repos/straight.el/bootstrap.el"
        (or (bound-and-true-p straight-base-dir)
            user-emacs-directory)))
      (bootstrap-version 7))
  (unless (file-exists-p bootstrap-file)
    (with-current-buffer
        (url-retrieve-synchronously
         "https://raw.githubusercontent.com/radian-software/straight.el/develop/install.el"
         'silent 'inhibit-cookies)
      (goto-char (point-max))
      (eval-print-last-sexp)))
  (load bootstrap-file nil 'nomessage))
(straight-use-package 'use-package)
#+end_src

* Emacs
This section contains basic configuration which does not depend on any packages.

#+begin_src emacs-lisp
(use-package emacs
  :custom
  ;; Scratch Buffer
  (initial-major-mode 'org-mode)
  (initial-scratch-message "")
  (inhibit-startup-screen t)

  ;; Don't use mouse
  (use-dialog-box nil)

  ;; Require confirmation when killing emacs
  (confirm-kill-emacs 'y-or-n-p)

  ;; Set custom file location to a temp file, we don't want it
  (custom-file (make-temp-file "emacs-custom-"))

  ;; Use visible bell instead of audible bell
  (visible-bell t)

  ;; Don't resize frame, let the WM do this
  (frame-resize-pixelwise t)
  (frame-inhibit-implied-resize t)

  ;; Scrolling
  (scroll-error-top-bottom t)
  (scroll-preserve-screen-position t)

  ;; Use view-mode for read-only buffers
  (view-read-only t)

  ;; Put all backups in .saves instead of the original directory
  (backup-directory-alist `(("." . ,(concat user-emacs-directory "backups"))))

  ;; Backup by copying is safer for e.g. links
  (backup-by-copying t)

  ;; Keep some numbered versions with version control
  (delete-old-versions t)
  (kept-new-versions 6)
  (kept-old-versions 2)
  (version-control t)

  ;; Determine how time is displayed
  (display-time-default-load-average nil)
  (display-time-day-and-date t)
  (display-time-24hr-format t)

  ;; Sentences can end with a single space
  (sentence-end-double-space nil)

  ;; Tab bar
  (tab-bar-new-button-show nil)
  (tab-bar-close-button-show nil)

  ;; Uniquification
  (uniquify-buffer-name-style 'forward)

  ;; Don't put hl-line on every window with buffer
  (hl-line-sticky-flag nil)

  ;; Diffs should be read only
  (diff-default-read-only t)

  :init
  ;; Use 4 spaces instead of tabs
  (setq-default indent-tabs-mode nil)
  (setq-default tab-width 4)

  ;; Disregard narrowing for line numbers
  (setq-default display-line-numbers-widen t)

  :config
  ;; Make native compilation silent.
  (when (native-comp-available-p)
    (setq native-comp-async-report-warnings-errors 'silent))

  ;; Load ancillary configuration files
  (org-babel-load-file (locate-user-emacs-file "my.org"))

  ;; Show misc info in modeline
  (which-function-mode 1)
  (display-battery-mode 1)
  (display-time-mode)

  ;; Enable which-key-mode
  (which-key-mode)

  ;; Disable keybinding for goal columns
  (global-unset-key (kbd "C-x C-n"))
  :general-config
  ("C-c B" 'ibuffer
   "C-c l" 'display-line-numbers-mode
   "M-o" 'other-window
   "M-O" 'other-frame
   "M-p" 'scroll-down-line
   "M-n" 'scroll-up-line
   "C-M-S-v" 'scroll-other-window-down))

    "Return t if the sun is currently up, otherwise nil."
    (let* ((solar-data (solar-sunrise-sunset (calendar-current-date)))
           (sunrise-decimal (car (car solar-data)))
           (sunset-decimal (car (nth 1 solar-data)))
           (current-time-decimal (+ (string-to-number (format-time-string "%H"))
                                    (/ (string-to-number (format-time-string "%M"))
                                       60.0))))
      (and (> current-time-decimal sunrise-decimal)
           (< current-time-decimal sunset-decimal))))

*** Commands
**** Open bookmark
Open a bookmark by ~cd~'ing to it or ~find-file~'ing it.

stolen from https://gist.github.com/kkatsuyuki/fa930411a86169c3bb1f03337d4af280
#+begin_src emacs-lisp
(defun pcomplete/eshell-mode/bm ()
  "Completion for `bm'"
  (pcomplete-here (bookmark-all-names)))

(defun eshell/bm (&rest args)
  "Open a bookmark entry."
  (setq args (flatten-tree args))
  (let ((bookmark (car args)) filename)
    (if (setq filename (bookmark-get-filename bookmark))
        ;; If it points to a directory, change to it.
        (if (file-directory-p filename)
            (eshell/cd filename)
          ;; otherwise, just jump to the bookmark
          (bookmark-jump bookmark))
      (error "%s is not a bookmark" bookmark))))
#+end_src
