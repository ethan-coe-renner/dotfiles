    # --- VI Mode ---
    set -g fish_key_bindings fish_vi_key_bindings
    # Emulates vim's cursor shape behavior
    # Set the normal and visual mode cursors to a block
    set fish_cursor_default block
    # Set the insert mode cursor to a line
    set fish_cursor_insert line
    # Set the replace mode cursors to an underscore
    set fish_cursor_replace_one underscore
    set fish_cursor_replace underscore

    set -gx EDITOR nvim
    set -gx LEDGER_FILE ~/git/finance/current.journal

    zoxide init fish | source
    starship init fish | source

# If this is a local terminal (not an ssh connection) set the prefix
# to C-a and move the status bar to the top. This way, we can run
# tmux in ssh connections without conflict.
if-shell "test -z $SSH_CONNECTION" "unbind C-b"
if-shell "test -z $SSH_CONNECTION" "set-option -g prefix C-a"
if-shell "test -z $SSH_CONNECTION" "set-option -g status-position top"
